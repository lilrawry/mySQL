CREATE TABLE livres (livre_id NUMBER GENERATED BY DEFAULT AS IDENTITY,
titre varchar2(100) NOT NULL,
auteur VARCHAR2(100) NOT NULL,
genre  VARCHAR2(100),
date_publication DATE,
PRIMARY KEY (livre_id) );

CREATE TABLE clients ( client_id NUMBER GENERATED ALWAYS AS IDENTITY,
nom  VARCHAR2(100) NOT NULL,
prenom  VARCHAR2(100) NOT NULL,
email  VARCHAR2(100) NOT NULL,
adresse  VARCHAR2(100),
UNIQUE  (email),
PRIMARY KEY (client_id)
);


CREATE TABLE Commandes (COMMANDE_ID NUMBER GENERATED ALWAYS AS IDENTITY,
DATE_COMMANDE DATE NOT NULL,
MONTANT_TOTAL NUMBER(2),
CLIENT_ID NUMBER,
CONSTRAINT COMMANDES_MONTANT_TOTAL_CK CHECK (MONTANT_TOTAL>0),
PRIMARY KEY (COMMANDE_ID)
);
